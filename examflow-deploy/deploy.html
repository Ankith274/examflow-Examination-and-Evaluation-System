create-s3-bucket.sh
BASH
1# ── STEP 2: Create your S3 bucket ──
2
3# Create the bucket (ap-south-1 = India/Mumbai region)
4aws s3api create-bucket \
5 --bucket my-bca-project-2026 \
6 --region ap-south-1 \

10aws s3 website s3://my-bca-project-2026 \

12 --error-document error.html

15aws s3api put-public-access-block \
16 --bucket my-bca-project-2026 \
17 --public-access-block-configuration \
18 BlockPublicAcls=false,IgnorePublicAcls=false,\
19 BlockPublicPolicy=false,RestrictPublicBuckets=false
20
21# Add bucket policy to allow public read access
22aws s3api put-bucket-policy \
23 --bucket my-bca-project-2026 \
24 --policy file://bucket-policy.json

🚀 Complete AWS S3 Deploy Script (One File)
deploy-to-aws-s3.sh
BASH
1#!/bin/bash
2# ════════════════════════════════════════════════
3# AWS S3 Static Website Deployment Script
4# BCA Final Year Project — KLH University 2026
5# ════════════════════════════════════════════════
6
7# ── CONFIGURATION ──
8BUCKET_NAME="my-bca-project-2026" # Change this!
9REGION="ap-south-1" # India/Mumbai
10SOURCE_DIR="./build" # Your project folder
11INDEX_DOC="index.html"
12ERROR_DOC="error.html"
13
14# Colors for output
15GREEN='\033[0;32m'; RED='\033[0;31m'; NC='\033[0m'
16
17echo "🚀 Starting AWS S3 Deployment..."
18
19# ── STEP 1: Check AWS CLI is installed ──
20if ! command -v aws &>/dev/null; then
21 echo "${RED}❌ AWS CLI not installed. Install from aws.amazon.com/cli${NC}"
22 exit 1
23fi
24
25# ── STEP 2: Create bucket if it doesn't exist ──
26echo "📦 Creating S3 bucket: $BUCKET_NAME"
27aws s3api create-bucket \
28 --bucket "$BUCKET_NAME" \
29 --region "$REGION" \
30 --create-bucket-configuration \
31 LocationConstraint="$REGION" 2>/dev/null || true
32
33# ── STEP 3: Configure for static website hosting ──
34echo "🌐 Enabling website hosting..."
35aws s3api put-bucket-website \
36 --bucket "$BUCKET_NAME" \
37 --website-configuration '{
38 "IndexDocument":{"Suffix":"'$INDEX_DOC'"},
39 "ErrorDocument":{"Key":"'$ERROR_DOC'"}}'
40
41# ── STEP 4: Set public access ──
42echo "🔓 Setting public access..."
43aws s3api put-public-access-block \
44 --bucket "$BUCKET_NAME" \
45 --public-access-block-configuration \
46 BlockPublicAcls=false,IgnorePublicAcls=false,\
47 BlockPublicPolicy=false,RestrictPublicBuckets=false
48
49# ── STEP 5: Apply bucket policy ──
50echo "📋 Applying bucket policy..."
51aws s3api put-bucket-policy \
52 --bucket "$BUCKET_NAME" \
53 --policy '{"Version":"2012-10-17","Statement":[{
54 "Sid":"PublicRead","Effect":"Allow","Principal":"*",
55 "Action":"s3:GetObject",
56 "Resource":"arn:aws:s3:::'$BUCKET_NAME'/*"}]}'
57
58# ── STEP 6: Upload all files ──
59echo "📤 Uploading website files..."
60aws s3 sync "$SOURCE_DIR" s3://"$BUCKET_NAME"/ \
61 --delete \
62 --exclude "*.DS_Store" \
63 --exclude "*.git/*"
64
65# ── DONE ──
66echo "${GREEN}✅ DEPLOYMENT SUCCESSFUL!${NC}"
67echo "🌐 Your website URL:"
68echo " http://$BUCKET_NAME.s3-website.$REGION.amazonaws.com"
Save this as deploy.sh → run chmod +x deploy.sh → then ./deploy.sh to deploy!
AWS EC2 — Full Server Deployment

Deploy Full-Stack App to AWS EC2

Full virtual Linux server. Run Node.js, PHP, Python. Host databases. 750 hrs/month FREE with t2.micro. Perfect for BCA final year dynamic projects.
🏠›☁️ AWS›🖥️ EC2 Server
🖥️ EC2 Complete Setup Script
aws-ec2-setup.sh
BASH
1#!/bin/bash
2# ═══════════════════════════════════════════════
3# AWS EC2 — Full Server Setup for BCA Project
4# Run this AFTER SSH-ing into your EC2 instance
5# ═══════════════════════════════════════════════
6
7# ── 1. Launch EC2 Instance (AWS Console steps) ──
8# Go to: EC2 → Launch Instance
9# Choose: Amazon Linux 2 / Ubuntu 22.04 LTS
10# Type: t2.micro (FREE TIER)
11# Create key pair → download .pem file
12# Open ports: 22 (SSH), 80 (HTTP), 443 (HTTPS)
13
14# ── 2. Connect via SSH ──
15# From your computer (replace with your values):
16ssh -i "my-key.pem" ubuntu@"YOUR-EC2-PUBLIC-IP"
17
18# ── 3. Update server packages ──
19sudo apt update && sudo apt upgrade -y
20
21# ── 4. Install Node.js ──
22curl -fsSL https://deb.nodesource.com/setup_18.x | sudo -E bash -
23sudo apt install -y nodejs
24node --version # Should show v18.x
25npm --version # Should show 9.x
26
27# ── 5. Install Nginx web server ──
28sudo apt install nginx -y
29sudo systemctl start nginx
30sudo systemctl enable nginx # Auto-start on reboot
31
32# ── 6. Install MySQL (optional - for database) ──
33sudo apt install mysql-server -y
34sudo mysql_secure_installation # Set root password
35
36# ── 7. Clone your project from GitHub ──
37cd /var/www/html
38sudo git clone https://github.com/yourusername/bca-project.git
39cd bca-project
40npm install
41npm run build
42
43# ── 8. Install PM2 to keep Node running ──
44sudo npm install -g pm2
45pm2 start app.js --name "bca-project"
46pm2 startup # Auto-restart on reboot
47pm2 save
48
49# ── 9. Configure Nginx to proxy to Node ──
50sudo nano /etc/nginx/sites-available/bca-project
51
52# Add this Nginx config:
53server {
54 listen 80;
55 server_name YOUR-EC2-PUBLIC-IP;
56 location / {
57 proxy_pass http://localhost:3000;
58 proxy_http_version 1.1;
59 proxy_set_header Upgrade $http_upgrade;
60 proxy_set_header Connection 'upgrade';
61 }
62}
63
64sudo nginx -t # Test config
65sudo systemctl reload nginx
66
67# ✅ Visit: http://YOUR-EC2-PUBLIC-IP to see your site!

GitHub Actions — Automated CI/CD Pipeline

Auto-Deploy on Every Git Push

🔄 .github/workflows/deploy.yml — Full CI/CD Pipeline
.github/workflows/deploy.yml
YAML
1# ══════════════════════════════════════════════════
2# GitHub Actions CI/CD — Auto Deploy to AWS S3
3# Every git push → build → test → deploy automatically
4# ══════════════════════════════════════════════════
5
6name: Deploy BCA Project to AWS S3
7
8on:
9 push:
10 branches: [ main ] # Trigger on push to main
11 pull_request:
12 branches: [ main ]
13
14env:
15 AWS_REGION: ap-south-1
16 S3_BUCKET: my-bca-project-2026
17
18jobs:
19 build-and-deploy:
20 runs-on: ubuntu-latest
21
22 steps:
23 # Step 1: Get the code
24 - name: 📥 Checkout Code
25 uses: actions/checkout@v4
26
27 # Step 2: Setup Node.js
28 - name: ⚙️ Setup Node.js
29 uses: actions/setup-node@v4
30 with:
31 node-version: '18'
32 cache: 'npm'
33
34 # Step 3: Install & Build
35 - name: 📦 Install Dependencies
36 run: npm ci
37
38 - name: 🔨 Build Project
39 run: npm run build
40
41 - name: 🧪 Run Tests
42 run: npm test -- --watchAll=false
43 continue-on-error: true
44
45 # Step 4: Configure AWS
46 - name: 🔐 Configure AWS Credentials
47 uses: aws-actions/configure-aws-credentials@v4
48 with:
49 aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
50 aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
51 aws-region: ${{ env.AWS_REGION }}
52
53 # Step 5: Deploy to S3
54 - name: 🚀 Deploy to S3
55 run: |
56 aws s3 sync ./build s3://${{ env.S3_BUCKET }} \
57 --delete --exact-timestamps
58
59 # Step 6: Invalidate CloudFront cache
60 - name: 🔄 Invalidate CloudFront Cache
61 if: github.ref == 'refs/heads/main'
62 run: |
63 aws cloudfront create-invalidation \
64 --distribution-id ${{ secrets.CF_DISTRIBUTION_ID }} \
65 --paths "/*"
66
67 - name: ✅ Deployment Complete
68 run: echo "🎉 BCA Project deployed!"
Add AWS_ACCESS_KEY_ID and AWS_SECRET_ACCESS_KEY in GitHub → Settings → Secrets. Never put real keys in code!
